import {IComponent, IComponentFactory} from "./interfaces";
 
export {ISystem, System, MultiParallelSystem, MultiNonParallelSystem}
 
interface ISystem {
    process(factory:IComponentFactory<IComponent>, ...args: any[]);
    execute(...args: any[]);
}
 
 
 
// are generics need here ?
abstract class System implements ISystem {
    constructor() {}
    process(factory: IComponentFactory<IComponent>) {
        let l = factory.iterationLength;
        let f = factory.values;
        for (let i = 0; i < l; ++i) {
            if(f[i].active) {
                this.execute(f[i]);                
            }
        };
    }
    abstract execute(component:IComponent);
}
